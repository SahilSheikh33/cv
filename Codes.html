def list_operations(): 
    my_list = [1, 2, 3, 4, 5]
 
    # Accessing elements 
    print("List Access:") 
        print(my_list[0])  # Output: 1 
        print(my_list[-1])  # Output: 5
    # Slicing
    print("List Slicing:")
    print(my_list[1:4])  # Output: [2, 3, 4]

    
    
    # Modification
    print("List Modification:")
    my_list.append(6)
    print(my_list)  # Output: [1, 2, 3, 4, 5, 6]
    
    new_elements = [7, 8]
    my_list.extend(new_elements)
    print(my_list)  # Output: [1, 2, 3, 4, 5, 6, 7, 8]
    
    my_list.insert(2, 9)
    print(my_list)  # Output: [1, 2, 9, 3, 4, 5, 6, 7, 8]
    
    my_list.remove(3)
    print(my_list)  # Output: [1, 2, 9, 4, 5, 6, 7, 8]
    
    popped_element = my_list.pop(4)
    print(popped_element)  # Output: 5
    print(my_list)  # Output: [1, 2, 9, 4, 6, 7, 8]
       my_list[3] = 10
    print(my_list)  # Output: [1, 2, 9, 10, 6, 7, 8]

def tuple_operations():
    my_tuple = (1, 2, 3, 4, 5)
  
    # Accessing elements
    print("\nTuple Access:")
    print(my_tuple[0])  # Output: 1
    print(my_tuple[-1])  # Output: 5
   
    # Slicing
    print("Tuple Slicing:")
    print(my_tuple[1:4])  # Output: (2, 3, 4)

def dict_operations():
    my_dict = {'a': 1, 'b': 2, 'c': 3}
    
    # Accessing values
    print("\nDictionary Access:")
    print(my_dict['a'])  # Output: 1
    print(my_dict.get('b'))  # Output: 2
    
    # Adding and Updating
    print("Dictionary Modification:")
    my_dict['d'] = 4
    print(my_dict)  # Output: {'a': 1, 'b': 2, 'c': 3, 'd': 4}
    
    my_dict.update({'e': 5, 'b': 10})
    print(my_dict)  # Output: {'a': 1, 'b': 10, 'c': 3, 'd': 4, 'e': 5}
    
    # Removing
    removed_value = my_dict.pop('a')
    print(removed_value)  # Output: 1
    print(my_dict)  # Output: {'b': 10, 'c': 3, 'd': 4, 'e': 5}
    
    # Clearing the dictionary
    my_dict.clear()
    print(my_dict)  # Output: {}
if __name__ == "__main__":
    list_operations()
    tuple_operations()
    dict_operations()
